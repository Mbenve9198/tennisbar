  const { locale } = useRouter()
  const [translations, setTranslations] = useState({})
  
  useEffect(() => {
    import(`../locales/${locale}/${namespace}.json`)
      .then(setTranslations)
  }, [locale, namespace])
  
  return { t: (key: string) => translations[key] || key }
}
